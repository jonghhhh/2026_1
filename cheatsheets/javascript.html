3문자열 (String)

let s = "Hello";
s.length           // 5
s[0]               // 'H'
s.toUpperCase()    // 'HELLO'
s.includes("ell") // true
s.startsWith("He")// true
s.slice(1,3)       // 'el'
s.split("")        // ['H','e',...]
s.replace("H","h")// 'hello'
s.trim()           // 양쪽 공백 제거
s.padStart(8,"0") // '000Hello'
s.repeat(2)       // 'HelloHello'

4배열 (Array)

let arr = [1,2,3,4,5];
arr.push(6)    // 끝에 추가
arr.pop()      // 끝에서 제거
arr.unshift(0) // 앞에 추가
arr.shift()    // 앞에서 제거
arr.slice(1,3) // [2,3] 복사
arr.splice(1,2)// 삭제+삽입
arr.indexOf(3) // 2
arr.includes(3)// true
arr.concat([6])// 합치기
arr.join("-")  // "1-2-3-4-5"
[...arr]        // 배열 복사

// 구조분해
let [a, b, ...rest] = arr;

5배열 고차함수

arr.forEach((v,i) => { ... })
arr.map(x => x * 2)
  // [2,4,6,8,10]
arr.filter(x => x > 2)
  // [3,4,5]
arr.reduce((acc,x) => acc+x, 0)
  // 15 (합계)
arr.find(x => x > 3)
  // 4 (첫 매칭)
arr.findIndex(x => x > 3)
  // 3
arr.some(x => x > 4)  // true
arr.every(x => x > 0) // true
arr.sort((a,b) => a-b)
arr.flat()  // 중첩 해제
arr.flatMap(x => [x, x*2])

6객체 (Object)

const obj = {
  name: "Kim", age: 30,
  greet() { return `Hi ${this.name}`; }
};
obj.name          // 'Kim'
obj["age"]        // 30
obj.email = "a@b" // 추가
delete obj.age     // 삭제

Object.keys(obj)   // ['name',...]
Object.values(obj) // ['Kim',...]
Object.entries(obj)// [['name','Kim']]
Object.assign({},obj) // 복사
{...obj}           // 전개 복사

// 구조분해
const {name, age} = obj;

9비동기 (Promise / async)

// Promise
fetch("/api/data")
  .then(res => res.json())
  .then(data => console.log(data))
  .catch(err => console.error(err));

// async/await (권장)
async function getData() {
  try {
    const res = await fetch("/api");
    const data = await res.json();
    return data;
  } catch (err) {
    console.error(err);
  }
}
