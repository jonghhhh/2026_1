1이미지 읽기 / 쓰기 / 표시

import cv2
import numpy as np

# 읽기
img = cv2.imread("img.jpg")
  # IMREAD_COLOR (기본, BGR)
  # IMREAD_GRAYSCALE (흑백)
  # IMREAD_UNCHANGED (알파)
gray = cv2.imread("img.jpg", 0)

# 표시
cv2.imshow("윈도우", img)
cv2.waitKey(0)
cv2.destroyAllWindows()

# 저장
cv2.imwrite("out.jpg", img)
⚠️ OpenCV는 RGB가 아닌 BGR 색상!

2속성 & 색상 변환

img.shape   # (높이, 너비, 채널)
img.size    # 전체 픽셀 수
img.dtype   # uint8
h, w, c = img.shape

# 색상 변환
rgb  = cv2.cvtColor(img,
  cv2.COLOR_BGR2RGB)
gray = cv2.cvtColor(img,
  cv2.COLOR_BGR2GRAY)
hsv  = cv2.cvtColor(img,
  cv2.COLOR_BGR2HSV)

# 채널 분리/합치기
b, g, r = cv2.split(img)
merged = cv2.merge([b, g, r])

3크기 조절 & 기하 변환

# 리사이즈
resized = cv2.resize(img,
  (400,300))  # (너비,높이)
resized = cv2.resize(img, None,
  fx=0.5, fy=0.5)  # 비율

# 자르기 (NumPy 슬라이싱)
crop = img[100:300, 50:250]
  # img[y1:y2, x1:x2]

# 회전
center = (w//2, h//2)
M = cv2.getRotationMatrix2D(
  center, 45, 1.0)
rot = cv2.warpAffine(img,M,(w,h))

# 뒤집기
cv2.flip(img, 1)
  # 0:상하 1:좌우 -1:둘다

4그리기 (Drawing)

# 선
cv2.line(img, (0,0), (200,200),
  (0,255,0), 2)  # BGR, 두께

# 사각형
cv2.rectangle(img,
  (50,50), (200,200),
  (255,0,0), 2)  # -1:채움

# 원
cv2.circle(img,
  (100,100), 50,
  (0,0,255), -1)

# 텍스트
cv2.putText(img, "Hello",
  (10,50),
  cv2.FONT_HERSHEY_SIMPLEX,
  1, (255,255,255), 2)

5필터 & 블러

# 가우시안 블러
blur = cv2.GaussianBlur(img,
  (5,5), 0)

# 미디언 블러 (노이즈 제거)
med = cv2.medianBlur(img, 5)

# 양방향 필터 (에지 보존)
bi = cv2.bilateralFilter(
  img, 9, 75, 75)

# 사용자 정의 커널
kernel = np.ones((3,3)) / 9
filtered = cv2.filter2D(
  img, -1, kernel)

# 샤프닝 커널
sharp = np.array([
  [0,-1,0],[-1,5,-1],[0,-1,0]])

6이진화 & 임계값

# 전역 임계값
ret, thresh = cv2.threshold(
  gray, 127, 255,
  cv2.THRESH_BINARY)
  # THRESH_BINARY_INV
  # THRESH_TRUNC
  # THRESH_OTSU (자동)

# 적응형 임계값
adap = cv2.adaptiveThreshold(
  gray, 255,
  cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
  cv2.THRESH_BINARY, 11, 2)

# Otsu 자동 이진화
_, otsu = cv2.threshold(
  gray, 0, 255,
  cv2.THRESH_BINARY + cv2.THRESH_OTSU)

7에지 검출 & 모폴로지

# Canny 에지
edges = cv2.Canny(gray, 100, 200)

# Sobel / Laplacian
sx = cv2.Sobel(gray,cv2.CV_64F,1,0)
sy = cv2.Sobel(gray,cv2.CV_64F,0,1)
lap = cv2.Laplacian(gray,cv2.CV_64F)

# 모폴로지 연산
k = cv2.getStructuringElement(
  cv2.MORPH_RECT, (5,5))
erode = cv2.erode(img, k, iterations=1)
dilate = cv2.dilate(img, k)
opening = cv2.morphologyEx(
  img, cv2.MORPH_OPEN, k)
closing = cv2.morphologyEx(
  img, cv2.MORPH_CLOSE, k)

8컨투어 (Contour)

# 컨투어 찾기
contours, hier = cv2.findContours(
  thresh, cv2.RETR_TREE,
  cv2.CHAIN_APPROX_SIMPLE)

# 컨투어 그리기
cv2.drawContours(img,
  contours, -1, (0,255,0), 2)

# 컨투어 속성
area = cv2.contourArea(cnt)
peri = cv2.arcLength(cnt, True)
x,y,w,h = cv2.boundingRect(cnt)

# 근사 & 볼록 껍질
approx = cv2.approxPolyDP(
  cnt, 0.02*peri, True)
hull = cv2.convexHull(cnt)

9히스토그램 & 색상 분석

# 히스토그램 계산
hist = cv2.calcHist(
  [img], [0], None,
  [256], [0,256])

# 히스토그램 평활화
eq = cv2.equalizeHist(gray)

# CLAHE (적응형 평활화)
clahe = cv2.createCLAHE(
  clipLimit=2.0,
  tileGridSize=(8,8))
cl = clahe.apply(gray)

# HSV 기반 색상 마스킹
lower = np.array([35,50,50])
upper = np.array([85,255,255])
mask = cv2.inRange(hsv,lower,upper)
result = cv2.bitwise_and(
  img, img, mask=mask)

10비디오 처리

# 비디오/카메라 열기
cap = cv2.VideoCapture(0)  # 웹캠
cap = cv2.VideoCapture("v.mp4")

while cap.isOpened():
    ret, frame = cap.read()
    if not ret: break
    cv2.imshow("Video", frame)
    if cv2.waitKey(1) & 0xFF \
       == ord('q'): break
cap.release()

# 비디오 저장
fourcc = cv2.VideoWriter_fourcc(
  *'mp4v')
out = cv2.VideoWriter('out.mp4',
  fourcc, 30, (w,h))
out.write(frame)
out.release()

12얼굴/객체 검출 & DNN

# Haar Cascade 얼굴 검출
face_cas = cv2.CascadeClassifier(
  cv2.data.haarcascades +
  'haarcascade_frontalface_default.xml')
faces = face_cas.detectMultiScale(
  gray, 1.3, 5)
for (x,y,w,h) in faces:
  cv2.rectangle(img,
    (x,y),(x+w,y+h),(0,255,0),2)

# DNN 모델 로드
net = cv2.dnn.readNetFromCaffe(
  "deploy.prototxt",
  "model.caffemodel")
blob = cv2.dnn.blobFromImage(
  img, 1.0, (300,300))
net.setInput(blob)
out = net.forward()
