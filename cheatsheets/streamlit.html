# 기본 구조 (app.py)
import streamlit as st
import pandas as pd

st.set_page_config(
  page_title="My App",
  page_icon="🎈",
  layout="wide",
  # "centered" (기본)
)

st.title("Hello Streamlit!")

4입력 위젯 — 버튼/텍스트

# 버튼
if st.button("클릭"):
    st.write("클릭됨!")

# 텍스트 입력
name = st.text_input("이름",
  value="기본값")
text = st.text_area("내용",
  height=200)

# 숫자 입력
n = st.number_input("숫자",
  min_value=0, max_value=100,
  value=50, step=1)

# 날짜/시간
d = st.date_input("날짜")
t = st.time_input("시간")

# 파일 업로드
f = st.file_uploader("파일",
  type=["csv","xlsx"])

5입력 위젯 — 선택/슬라이더

# 셀렉트박스
opt = st.selectbox("선택",
  ["A", "B", "C"])

# 멀티셀렉트
opts = st.multiselect("다중선택",
  ["사과","바나나","체리"])

# 라디오
r = st.radio("성별",
  ["남", "여"])

# 체크박스
if st.checkbox("동의"):
    st.write("동의함")

# 슬라이더
val = st.slider("값",
  0, 100, 50)
rng = st.slider("범위",
  0.0, 1.0, (0.2,0.8))

# 토글
on = st.toggle("활성화")

6레이아웃

# 사이드바
with st.sidebar:
    st.title("사이드바")
    opt = st.radio("메뉴", ["A","B"])

# 컬럼
col1, col2 = st.columns(2)
with col1:
    st.write("왼쪽")
with col2:
    st.write("오른쪽")
# 비율 지정: st.columns([3,1,1])

# 탭

with t1: st.write("탭1 내용")

# 확장
with st.expander("더보기"):
    st.write("숨겨진 내용")

# 컨테이너

    st.write("그룹")

9Session State (상태 관리)

# 초기화
if "count" not in st.session_state:
    st.session_state.count = 0

# 증가 버튼
if st.button("증가"):
    st.session_state.count += 1

st.write(st.session_state.count)

# 콜백 함수
def on_change():
    st.session_state.result = \
      st.session_state.input * 2

st.number_input("값",
  key="input",
  on_change=on_change)

11폼 & 채팅

# 폼 (한번에 제출)
with st.form("my_form"):
    name = st.text_input("이름")
    age = st.number_input("나이")
    submitted = st.form_submit_button(
      "제출")
    if submitted:
        st.write(name, age)

# 채팅 인터페이스
msg = st.chat_input("메시지")
if msg:
    with st.chat_message("user"):
        st.write(msg)
    with st.chat_message("assistant"):
        st.write("응답...")
