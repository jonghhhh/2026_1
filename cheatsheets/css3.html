1선택자 (Selectors)

*         /* 전체 */
div       /* 태그 */
.class    /* 클래스 */
#id       /* ID */
div p     /* 자손 */
div > p   /* 직계 자식 */
h1 + p   /* 인접 형제 */
h1 ~ p   /* 일반 형제 */
[attr]     /* 속성 존재 */
[attr="val"] /* 속성=값 */
[attr^="v"]  /* 시작 */
[attr$="v"]  /* 끝 */
[attr*="v"]  /* 포함 */

6Display & Position

display: block;
  /* inline | inline-block |
     none | flex | grid */
visibility: hidden; /* 자리 유지 */

position: static;   /* 기본 */
position: relative; /* 원래위치 기준 */
position: absolute; /* 부모 기준 */

7Flexbox

.container {
  display: flex;
  flex-direction: row;
    /* column row-reverse */
  justify-content: center;
    /* start end space-between
       space-around space-evenly */
  align-items: center;
    /* start end stretch */
  flex-wrap: wrap;
  gap: 10px;
}
.item {
  flex: 1;  /* grow shrink basis */
  order: 2;
  align-self: flex-end;
}

8Grid

.container {
  display: grid;
  grid-template-columns:
    repeat(3, 1fr);
    /* 200px auto 1fr */
  grid-template-rows:
    100px auto;
  gap: 10px 20px;
  place-items: center;
}
.item {
  grid-column: 1 / 3; /* 병합 */
  grid-row: 1 / 2;
}

9변환 & 애니메이션

/* Transform */
transform: translateX(50px);
transform: rotate(45deg);
transform: scale(1.5);
transform: skewX(10deg);

/* Transition */
transition: all 0.3s ease;
  /* ease-in ease-out
     ease-in-out linear */

/* Animation */
@keyframes slide {
  from { transform:translateX(0); }
  to   { transform:translateX(100px); }
}
animation: slide 2s ease
  infinite alternate;

11반응형 (Media Query)

@media (max-width: 768px) {
  .container {
    flex-direction: column;
  }
}
@media (min-width: 1024px) {
  .sidebar {
    display: block;
  }
}
@media print {
  .no-print { display:none; }
}
/* 브레이크포인트 관례 */
/* 모바일: ~480px  태블릿: ~768px
   데스크탑: ~1024px+  */

12CSS 변수 & 기타

/* CSS 변수 */
:root {
  --primary: #264de4;
  --gap: 16px;
}
.btn {
  color: var(--primary);
  padding: var(--gap);
}

/* 우선순위 (높→낮) */
/* !important > inline style >
   #id > .class > tag > * */

/* calc() */
width: calc(100% - 200px);

/* clamp() */
font-size: clamp(14px,2vw,20px);
