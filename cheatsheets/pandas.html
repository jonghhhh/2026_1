import pandas as pd | 2026 Updated

1데이터 구조 생성

# Series (1D)
s = pd.Series([3,-5,7],
  index=['a','b','c'])

# DataFrame (2D)
data = {'이름': ['김','이','박'],
        '나이': [25,30,28],
        '점수': [90,85,95]}
df = pd.DataFrame(data)
#    이름 나이 점수
# 0  김   25   90
# 1  이   30   85
# 2  박   28   95

# CSV
df = pd.read_csv('f.csv',

df.to_csv('out.csv', index=False)

# Excel
df = pd.read_excel('f.xlsx',
  sheet_name='Sheet1')
df.to_excel('out.xlsx')

# JSON
df = pd.read_json('f.json')

# SQL
df = pd.read_sql(query, engine)

3기본 정보 확인

메서드설명
df.head(n) .tail(n)처음/마지막 n행
df.shape(행수, 열수)
df.info()타입, 널 요약
df.describe()기술통계
df.dtypes열별 데이터타입
df.columns / .index열명 / 인덱스
df.count() .nunique()비결측/고유값 수
df['c'].value_counts()값별 빈도

4선택 & 인덱싱

# 열 선택
df['이름']            # Series
df[['이름','나이']]   # DataFrame

# loc: 라벨 기반
df.loc[0,'이름']
df.loc[0:2,['이름','점수']]

# iloc: 위치(정수) 기반
df.iloc[0,0]
df.iloc[0:2, 0:2]

# 불리언 인덱싱
df[df['나이'] > 25]
df[(df['나이']>25) & (df['점수']>85)]
df[df['이름'].isin(['김','박'])]
df.query('나이>25 and 점수>85')

5데이터 정제 (결측/중복)

# 결측치
df.isnull().sum()       # 열별 결측
df.dropna()             # 결측 행 삭제
df.dropna(axis=1)      # 결측 열 삭제
df.fillna(0)            # 0으로 채움
df.fillna(method='ffill')# 앞값
df.fillna(df.mean())   # 평균

# 중복
df.duplicated()
df.drop_duplicates()

# 타입 변환
df['나이'].astype(int)
# 이름 변경 / 값 치환
df.rename(columns={'나이':'age'})
df.replace({'김':'Kim'})

6열 추가/삭제/변환

# 추가
df['등급'] = ['A','B','A']
df['비율'] = df['점수']/df['나이']

# 삭제
df.drop(columns=['등급'])
df.drop([0,2])  # 행 삭제

# apply (함수 적용)
df['점수'].apply(lambda x: x*1.1)
df.apply(lambda x: x.max()-x.min())

# map (Series 매핑)
df['등급'].map({'A':4,'B':3})

7정렬 & 통계

# 정렬
df.sort_values('나이')
df.sort_values('점수',ascending=False)
df.sort_values(['등급','점수'])
df.nlargest(3,'점수')  # 상위3
통계 메서드설명
.sum() .mean()합계 / 평균
.median() .std()중앙값/표준편차
.min() .max()최소 / 최대
.cumsum()누적합
.corr()상관행렬
.describe()기술통계 전체
.rank()순위

8GroupBy (그룹 연산)

g = df.groupby('등급')
g.mean()          # 그룹별 평균
g.size()          # 그룹별 크기
g.describe()      # 기술통계

# 다중 키
df.groupby(['등급','성별']).mean()

# agg: 다양한 집계
g.agg({'점수':['mean','max'],
      '나이':'min'})

# transform: 원래 크기 유지
g['점수'].transform(
  lambda x:(x-x.mean())/x.std())

# filter: 조건 그룹만
g.filter(lambda x: x['점수'].mean()>85)

9병합 & 결합

# merge (SQL JOIN)
pd.merge(df1, df2, on='key')
pd.merge(df1, df2,
  left_on='a', right_on='b',
  how='left')
# how: inner outer left right

# concat (단순 결합)
pd.concat([df1, df2])       # 세로
pd.concat([df1,df2],axis=1)# 가로

# join (인덱스 기반)
df1.join(df2, how='outer')

10피벗 & 리셰이프

# pivot_table
pd.pivot_table(df,
  values='점수', index='등급',
  columns='성별', aggfunc='mean')

# melt (wide → long)
pd.melt(df, id_vars=['이름'],
  value_vars=['국어','영어'])

# stack/unstack
df.stack()    # 열→행
df.unstack()  # 행→열

# 더미변수
pd.get_dummies(df['등급'])

12날짜/시간 (.dt 접근자)

df['날짜'] = pd.to_datetime(df['날짜'])
df['년']  = df['날짜'].dt.year
df['월']  = df['날짜'].dt.month
df['요일'] = df['날짜'].dt.day_name()

# 날짜 범위
pd.date_range('2026-01-01',
  periods=30, freq='D')

# 리샘플링
df.resample('M').mean()   # 월별
df.resample('W').sum()    # 주별
