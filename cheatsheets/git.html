🌿 Git Cheat Sheet

Git | 2026 Updated
Working Dir→ git add → Staging Area→ git commit → Local Repo→ git push → Remote Repo

1설정 & 초기화

# 사용자 설정
git config --global user.name "이름"
git config --global user.email "email"
git config --list     # 설정 확인

# 저장소 생성
git init              # 새 저장소
git clone url         # 복제
git clone url folder # 폴더명 지정

2상태 확인 & 변경사항

git status           # 상태 확인
git diff             # 변경사항 비교
git diff --staged    # 스테이지 비교
git diff branch1 branch2

# 스테이징
git add file.txt     # 파일 추가
git add .            # 전체 추가
git add -p file     # 부분 추가
git reset file       # 언스테이지

3커밋

git commit -m "메시지"
git commit -a -m "msg"
  # tracked 파일 자동 add+commit
git commit --amend
  # 마지막 커밋 수정
git commit --amend -m "새메시지"
⚠️ 커밋 메시지: 50자 이내, 현재시제, 명령형으로 작성 (예: "Add login feature")

4로그 & 히스토리

git log              # 전체 로그
git log --oneline    # 한줄 요약
git log --graph      # 그래프
git log --oneline --graph --all
git log -p file     # 파일 변경이력
git log --since="2w" # 2주내
git show commit_id  # 커밋 상세
git blame file      # 라인별 작성자
git reflog           # HEAD 이력

5브랜치

git branch           # 목록
git branch name     # 생성
git checkout name   # 전환
git switch name     # 전환 (신규)
git checkout -b name # 생성+전환
git switch -c name  # 생성+전환
git branch -d name  # 삭제
git branch -D name  # 강제삭제
git branch -m old new # 이름변경
git branch -a       # 전체(원격포함)

6병합 & 리베이스

# merge (현재 브랜치에 병합)
git merge branch

# 충돌 해결 후
git add resolved-file
git commit

# rebase (히스토리 정리)
git rebase main
git rebase --abort    # 취소
git rebase --continue # 계속
git rebase -i HEAD~3  # 대화형
⚠️ 공유된 브랜치에서는 rebase 금지!

7원격 저장소 (Remote)

git remote -v         # 원격 목록
git remote add origin url
git remote remove origin

# 가져오기 & 보내기
git fetch origin     # 다운로드만
git pull origin main # fetch+merge
git pull --rebase    # fetch+rebase
git push origin main # 업로드
git push -u origin main
  # 업스트림 설정+push
git push --tags      # 태그 전송

8되돌리기 (Undo)

# 작업 디렉토리 변경 취소
git checkout -- file
git restore file    # 신규

# 스테이지 취소
git restore --staged file

# 커밋 되돌리기
git revert commit   # 새 커밋 생성

# 히스토리 리셋 (주의!)
git reset --soft HEAD~1
  # 커밋 취소, 스테이지 유지
git reset --mixed HEAD~1
  # 커밋+스테이지 취소
git reset --hard HEAD~1
  # 모든 변경 삭제 (위험!)

9스태시 (Stash)

git stash            # 임시 저장
git stash save "msg"# 메시지 포함
git stash list       # 목록
git stash pop        # 복원+삭제
git stash apply      # 복원 (유지)
git stash drop       # 삭제
git stash clear      # 전체 삭제

10태그 & 릴리스

git tag              # 태그 목록
git tag v1.0         # 경량 태그
git tag -a v1.0 -m "msg"
  # 주석 태그 (권장)
git tag -d v1.0     # 태그 삭제
git show v1.0       # 태그 상세
git push origin v1.0
git push --tags      # 전체 태그

11.gitignore

# .gitignore 파일 예시
*.log              # 모든 .log 파일
node_modules/      # 폴더 전체
.env               # 환경변수
__pycache__/       # 파이썬 캐시
*.pyc              # 컴파일 파일
.DS_Store          # macOS
dist/              # 빌드 출력
!important.log     # 예외 (추적)

# 이미 추적중인 파일 무시
git rm --cached file

12유용한 명령어

# Cherry-pick (특정 커밋 가져오기)
git cherry-pick commit_id

# Bisect (버그 커밋 찾기)
git bisect start
git bisect bad   # 현재=버그
git bisect good commit

# 파일 삭제/이동
git rm file
git mv old new

# 단축 명령 설정
git config --global alias.co checkout
git config --global alias.br branch
git config --global alias.st status
git config --global alias.lg \
  "log --oneline --graph --all"
