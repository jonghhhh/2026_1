1기본 구조 (Figure & Axes)

import matplotlib.pyplot as plt
import numpy as np

# 방법 1: 간단한 방식
plt.plot(x, y)
plt.show()

# 방법 2: OOP 방식 (권장)
fig, ax = plt.subplots(
    figsize=(10,6))
ax.plot(x, y)
plt.show()

# 방법 3: 여러 서브플롯
fig, axes = plt.subplots(
    nrows=2, ncols=2)
axes[0,0].plot(x, y)
axes[0,1].bar(x, y)
plt.tight_layout()
plt.show()

5제목 / 축 / 범례

# 제목 & 라벨
ax.set_title('제목', fontsize=14)
ax.set_xlabel('X축')
ax.set_ylabel('Y축')

# 축 범위
ax.set_xlim(0, 10)
ax.set_ylim(-1, 1)

# 눈금
ax.set_xticks([0,2,4,6,8])
ax.set_xticklabels(['a','b',...],
  rotation=45)

# 범례
ax.legend(loc='best')
ax.legend(loc='upper right')

# 그리드
ax.grid(True, alpha=0.3)

6텍스트 & 주석 & 선

# 텍스트
ax.text(5, 3, '텍스트',
  fontsize=12, ha='center',
  color='red', fontweight='bold')

# 화살표 주석
ax.annotate('최대값',
  xy=(5,10),        # 화살표 끝
  xytext=(6,12),    # 텍스트 위치
  arrowprops=dict(
    arrowstyle='->'))

# 수평/수직 참조선
ax.axhline(y=0,color='k',ls='--')
ax.axvline(x=5,color='r',ls=':')

# 영역 채우기
ax.fill_between(x,y1,y2,alpha=0.3)

9저장 & 닫기

# 저장 (show 전에 호출)
plt.savefig('chart.png',
  dpi=300,
  bbox_inches='tight',
  transparent=False)
plt.savefig('chart.pdf')
plt.savefig('chart.svg')

# 닫기/초기화
plt.cla()    # 축 초기화
plt.clf()    # 그림 초기화
plt.close()  # 창 닫기
plt.close('all') # 전체 닫기

11Pandas 직접 플롯

# DataFrame에서 바로
df.plot()               # 선그래프
df.plot(kind='bar')     # 막대
df.plot(kind='barh')    # 수평막대
df.plot(kind='scatter',
  x='col1', y='col2')  # 산점도
df['c'].plot(kind='hist')
df.plot(kind='box')
df.plot(kind='pie',y='c')

# Pandas + Matplotlib 조합
fig, ax = plt.subplots(
  figsize=(10,6))
df.plot(kind='bar', ax=ax)
ax.set_title('제목')
plt.savefig('out.png')
